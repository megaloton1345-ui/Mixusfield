local Util = {}

function Util.Safe(fn, ...)
    local ok, err = pcall(fn, ...)
    if not ok then warn("[Mixusfield] " .. tostring(err)) end
    return ok
end

function Util.DeepCopy(t)
    if type(t) ~= "table" then return t end
    local copy = {}
    for k, v in pairs(t) do copy[k] = Util.DeepCopy(v) end
    return copy
end

function Util.LerpColor(a, b, t)
    return Color3.new(
        a.R + (b.R - a.R) * t,
        a.G + (b.G - a.G) * t,
        a.B + (b.B - a.B) * t
    )
end

function Util.Clamp(v, min, max) return math.max(min, math.min(max, v)) end
function Util.Round(v, n) n = n or 1; return math.floor(v / n + 0.5) * n end
function Util.GetViewport() return workspace.CurrentCamera.ViewportSize end

function Util.Connect(signal, fn) return signal:Connect(fn) end

-- Glow gradient effect on a frame
function Util.ApplyGlow(frame, color, intensity)
    intensity = intensity or 0.7
    local grad = Util.Create("UIGradient", {
        Color = ColorSequence.new({
            ColorSequenceKeypoint.new(0, color or Color3.fromRGB(120,120,255)),
            ColorSequenceKeypoint.new(1, Color3.fromRGB(0,0,0)),
        }),
        Transparency = NumberSequence.new({
            NumberSequenceKeypoint.new(0, 1 - intensity),
            NumberSequenceKeypoint.new(1, 1),
        }),
        Rotation = 90,
    }, frame)
    return grad
end

-- Measure text bounds safely
function Util.MeasureText(text, size, font)
    local ok, result = pcall(function()
        return TextService:GetTextSize(
            text, size,
            font or Enum.Font.Gotham,
            Vector2.new(math.huge, math.huge)
        )
    end)
    return ok and result or Vector2.new(100, size)
end

-- HSV → Color3
function Util.HSVtoRGB(h, s, v)
    return Color3.fromHSV(h, s, v)
end

-- Color3 → HSV
function Util.RGBtoHSV(c)
    return Color3.toHSV(c)
end

function Util.GetDeviceType()
    local vp    = Util.GetViewport()
    local touch = UserInputService.TouchEnabled
    local w     = vp.X
    if touch then
        if w < 500  then return "Mobile"
        elseif w < 900 then return "Tablet"
        else return "TabletLarge" end
    else
        if w >= 2560 then return "Ultrawide"
        elseif w >= 1440 then return "Desktop"
        else return "Desktop" end
    end
end

function Util.IsPortrait()
    local vp = Util.GetViewport()
    return vp.Y > vp.X
end

function Util.Tween(inst, info, props)
    local t = TweenService:Create(inst, info, props)
    t:Play()
    return t
end

function Util.TI(t, style, dir)
    return TweenInfo.new(
        t or 0.25,
        style or Enum.EasingStyle.Quint,
        dir   or Enum.EasingDirection.Out
    )
end

function Util.Create(class, props, parent)
    local inst = Instance.new(class)
    for k, v in pairs(props or {}) do
        if k ~= "Parent" then pcall(function() inst[k] = v end) end
    end
    if parent then inst.Parent = parent end
    return inst
end

function Util.Pad(parent, top, right, bottom, left)
    return Util.Create("UIPadding", {
        PaddingTop    = UDim.new(0, top    or 0),
        PaddingRight  = UDim.new(0, right  or top or 0),
        PaddingBottom = UDim.new(0, bottom or top or 0),
        PaddingLeft   = UDim.new(0, left   or right or top or 0),
    }, parent)
end

function Util.Corner(parent, radius)
    return Util.Create("UICorner", { CornerRadius = UDim.new(0, radius or 8) }, parent)
end

function Util.Stroke(parent, color, thickness, transparency)
    return Util.Create("UIStroke", {
        Color           = color or Color3.fromRGB(255,255,255),
        Thickness       = thickness or 1,
        Transparency    = transparency or 0.85,
        ApplyStrokeMode = Enum.ApplyStrokeMode.Border,
    }, parent)
end

function Util.List(parent, dir, spacing, halign, valign)
    return Util.Create("UIListLayout", {
        FillDirection       = dir     or Enum.FillDirection.Vertical,
        Padding             = UDim.new(0, spacing or 6),
        HorizontalAlignment = halign  or Enum.HorizontalAlignment.Left,
        VerticalAlignment   = valign  or Enum.VerticalAlignment.Top,
        SortOrder           = Enum.SortOrder.LayoutOrder,
    }, parent)
end

local Responsive = {}

--[[
    Scale values for each breakpoint.
    Mobile:      touch-friendly, larger tap targets, no sidebar
    Tablet:      icon-only sidebar, slightly larger than mobile
    TabletLarge: small sidebar with labels
    Desktop:     full sidebar + search
    Ultrawide:   everything slightly bigger
]]
Responsive.Breakpoints = {
    Mobile      = { scale = 0.88, fontSize = 13, padding = 10, iconSize = 20, tapH = 48, topbarH = 52, tabBarH = 44, sidebarW = 0  },
    Tablet      = { scale = 0.93, fontSize = 13, padding = 11, iconSize = 20, tapH = 46, topbarH = 48, tabBarH = 40, sidebarW = 48 },
    TabletLarge = { scale = 0.97, fontSize = 13, padding = 12, iconSize = 21, tapH = 44, topbarH = 46, tabBarH = 38, sidebarW = 50 },
    Desktop     = { scale = 1.00, fontSize = 14, padding = 14, iconSize = 22, tapH = 38, topbarH = 44, tabBarH = 36, sidebarW = 46 },
    Ultrawide   = { scale = 1.10, fontSize = 15, padding = 16, iconSize = 24, tapH = 40, topbarH = 48, tabBarH = 38, sidebarW = 52 },
}

function Responsive:Get()
    return self.Breakpoints[Util.GetDeviceType()] or self.Breakpoints.Desktop
end

function Responsive:Scale(v)
    return math.round(v * self:Get().scale)
end

function Responsive:FontSize(base)
    local r = self:Get()
    return math.max(r.fontSize, math.round(base * r.scale))
end

function Responsive:Padding()   return self:Get().padding   end
function Responsive:IconSize()  return self:Get().iconSize  end
function Responsive:TapHeight() return self:Get().tapH      end
function Responsive:TopbarH()   return self:Get().topbarH   end
function Responsive:TabBarH()   return self:Get().tabBarH   end
function Responsive:SidebarW()  return self:Get().sidebarW  end
function Responsive:IsMobile()  return Util.GetDeviceType() == "Mobile" end
function Responsive:IsTouch()
    local d = Util.GetDeviceType()
    return d == "Mobile" or d == "Tablet" or d == "TabletLarge"
end

function Responsive:WindowSize()
    local vp     = Util.GetViewport()
    local device = Util.GetDeviceType()
    if device == "Mobile" then
        -- Leave visible margin from all edges (same look as reference screenshot)
        local w = math.min(vp.X - 20, 700)
        local h = math.min(vp.Y - 100, 460)
        return UDim2.new(0, w, 0, h)
    elseif device == "Tablet" then
        local w = math.min(vp.X - 40, 620)
        local h = math.min(vp.Y - 100, 430)
        return UDim2.new(0, w, 0, h)
    elseif device == "TabletLarge" then
        return UDim2.new(0, 680, 0, 460)
    elseif device == "Ultrawide" then
        return UDim2.new(0, 920, 0, 580)
    else
        return UDim2.new(0, 740, 0, 500)
    end
end

function Responsive:KeySystemSize()
    local vp     = Util.GetViewport()
    local device = Util.GetDeviceType()
    if device == "Mobile" then
        local w = math.min(vp.X - 16, 420)
        local h = math.min(vp.Y - 40, 460)
        return w, h
    elseif device == "Tablet" or device == "TabletLarge" then
        return 440, 400
    else
        return 460, 400
    end
end

function Responsive:NotifWidth()
    local vp = Util.GetViewport()
    if self:IsMobile() then
        return math.min(vp.X - 16, 320)
    end
    return 330
end

function Responsive:FabSize()
    -- Floating action button: larger on touch devices
    if self:IsTouch() then return 56 end
    return 48
end

function Themes:Set(name)
    local def = self.Definitions[name]
    if not def then warn("[Mixusfield] Theme not found: " .. tostring(name)); return end
    self.Name   = name
    self.Active = Util.DeepCopy(def)
    for _, fn in ipairs(self.Listeners) do Util.Safe(fn, self.Active) end
end

function Themes:OnChange(fn) table.insert(self.Listeners, fn) end
function Themes:Current()    return self.Active end

--[[
    Icons:Create — builds a clean icon container.

    Priority:
      1. AssetID type → ImageLabel with that exact ID
      2. Lucide/Feather type + name in LucideMap → ImageLabel (falls back silently on load fail)
      3. Everything else → TextLabel with SymbolMap lookup, clamped to 3 chars max
         NEVER produces "[]" — if lookup fails, uses "?" as fallback.

    Returns (container, innerObject)
    container = Frame sized to `size x size`
    innerObject = ImageLabel or TextLabel (use for Recolor)
]]
function Icons:Create(parent, iconDef, size, color)
    size  = size  or Responsive:IconSize()
    color = color or Themes:Current().Text

    local iconType = (iconDef and iconDef.Type)   or "Symbol"
    local iconSrc  = (iconDef and iconDef.Source) or "menu"

    local container = Util.Create("Frame", {
        Size                   = UDim2.new(0, size, 0, size),
        BackgroundTransparency = 1,
        Name                   = "IconContainer",
        ClipsDescendants       = false,
    }, parent)

    -- ── Helper: guaranteed-safe text label ─────────────────────────
    local function makeText(sym)
        -- Clamp to 3 chars, strip non-ASCII, never empty
        sym = tostring(sym or "")
        sym = sym:sub(1, 3)
        if #sym == 0 then sym = "?" end
        local lbl = Util.Create("TextLabel", {
            Size                   = UDim2.new(1, 0, 1, 0),
            BackgroundTransparency = 1,
            Text                   = sym,
            TextColor3             = color,
            TextScaled             = false,
            TextSize               = math.max(10, math.floor(size * 0.64)),
            Font                   = Enum.Font.GothamBold,
            TextXAlignment         = Enum.TextXAlignment.Center,
            TextYAlignment         = Enum.TextYAlignment.Center,
            RichText               = false,
        }, container)
        return lbl
    end

    -- ── Helper: ImageLabel with asset ID ───────────────────────────
    local function makeImage(assetId)
        local img = Util.Create("ImageLabel", {
            Size                   = UDim2.new(1, 0, 1, 0),
            BackgroundTransparency = 1,
            Image                  = tostring(assetId),
            ImageColor3            = color,
            ScaleType              = Enum.ScaleType.Fit,
        }, container)
        return img
    end

    -- ── Dispatch ───────────────────────────────────────────────────
    local iconSrcLower = tostring(iconSrc):lower()

    if iconType == "AssetID" or iconType == "Decal" then
        local src = tostring(iconSrc)
        if not src:find("rbxassetid://") and tonumber(src) then
            src = "rbxassetid://" .. src
        end
        return container, makeImage(src)

    elseif iconType == "Lucide" or iconType == "Feather" then
        local assetId = self.LucideMap[iconSrcLower]
            or self.LucideMap[iconSrc]
        if assetId then
            return container, makeImage(assetId)
        else
            -- No verified asset ID → clean text fallback
            local sym = self.SymbolMap[iconSrcLower]
                or self.SymbolMap[iconSrc]
                or tostring(iconSrc):sub(1, 1):upper()
            return container, makeText(sym)
        end

    elseif iconType == "Emoji" then
        local lbl = Util.Create("TextLabel", {
            Size                   = UDim2.new(1, 0, 1, 0),
            BackgroundTransparency = 1,
            Text                   = tostring(iconSrc),
            TextColor3             = color,
            TextScaled             = true,
            Font                   = Enum.Font.GothamBold,
            TextXAlignment         = Enum.TextXAlignment.Center,
            TextYAlignment         = Enum.TextYAlignment.Center,
        }, container)
        return container, lbl

    else
        -- Symbol or unknown: always safe ASCII
        local sym = self.SymbolMap[iconSrcLower]
            or self.SymbolMap[iconSrc]
            or tostring(iconSrc):sub(1, 2)
        if sym == "" then sym = "?" end
        return container, makeText(sym)
    end
end

-- Safe recolor — handles ImageLabel and TextLabel without error
function Icons:Recolor(iconObj, color, duration)
    if not iconObj then return end
    duration = duration or 0.18
    Util.Safe(function()
        if iconObj:IsA("ImageLabel") then
            Util.Tween(iconObj, Util.TI(duration), { ImageColor3 = color })
        elseif iconObj:IsA("TextLabel") then
            Util.Tween(iconObj, Util.TI(duration), { TextColor3 = color })
        end
    end)
end

local Anim = {}

function Anim.FadeIn(inst, duration, fromTrans)
    inst.BackgroundTransparency = fromTrans or 1
    Util.Tween(inst, Util.TI(duration or 0.3), { BackgroundTransparency = 0 })
end

function Anim.FadeOut(inst, duration, callback)
    local t = Util.Tween(inst, Util.TI(duration or 0.25), { BackgroundTransparency = 1 })
    if callback then t.Completed:Connect(callback) end
    return t
end

function Anim.Ripple(parent, x, y, color)
    color = color or Color3.fromRGB(255,255,255)
    local ripple = Util.Create("Frame", {
        Size                   = UDim2.new(0, 0, 0, 0),
        Position               = UDim2.new(0, x, 0, y),
        AnchorPoint            = Vector2.new(0.5, 0.5),
        BackgroundColor3       = color,
        BackgroundTransparency = 0.7,
        ZIndex                 = parent.ZIndex + 10,
    }, parent)
    Util.Corner(ripple, 9999)
    local sz = math.max(parent.AbsoluteSize.X, parent.AbsoluteSize.Y) * 2
    Util.Tween(ripple, Util.TI(0.5, Enum.EasingStyle.Quad), {
        Size                   = UDim2.new(0, sz, 0, sz),
        BackgroundTransparency = 1,
    }).Completed:Connect(function() ripple:Destroy() end)
end

function Anim.SlideOpen(inst, targetSize, duration)
    inst.Size = UDim2.new(targetSize.X.Scale, targetSize.X.Offset, 0, 0)
    inst.ClipsDescendants = true
    Util.Tween(inst, Util.TI(duration or 0.25, Enum.EasingStyle.Quint), { Size = targetSize })
end

function Anim.SlideClose(inst, duration, callback)
    local t = Util.Tween(inst, Util.TI(duration or 0.2, Enum.EasingStyle.Quint), {
        Size = UDim2.new(inst.Size.X.Scale, inst.Size.X.Offset, 0, 0)
    })
    if callback then t.Completed:Connect(callback) end
    return t
end

    Util.Safe(function()
        if readfile then
            local ok, content = pcall(readfile, self.Key .. ".json")
            if ok and content and content ~= "" then
                local decoded = HttpService:JSONDecode(content)
                self.Data = decoded or {}
            end
        end
    end)
end

function SaveSystem:Save()
    Util.Safe(function()
        if writefile then
            writefile(self.Key .. ".json", HttpService:JSONEncode(self.Data))
        end
    end)
end

function SaveSystem:Get(key, default)
    local v = self.Data[key]
    return v ~= nil and v or default
end

function SaveSystem:Set(key, value)
    self.Data[key] = value
    self:Save()
end

function NotifSystem:Push(config)
    config = config or {}
    if #self.Active >= self.MaxStack then
        table.insert(self.Queue, config)

    end

    local title    = config.Title   or "Notification"
    local content  = config.Content or ""
    local ntype    = config.Type    or "default"
    local duration = config.Duration or 4
    local theme    = Themes:Current()
    local accent   = theme[NotifColors[ntype] or "Accent"] or theme.Accent

    local notif = Util.Create("Frame", {
        Name                   = "Notification",
        Size                   = UDim2.new(1, 0, 0, 0),
        AutomaticSize          = Enum.AutomaticSize.Y,
        BackgroundColor3       = theme.Surface,
        BackgroundTransparency = 0.08,
        ZIndex                 = 200,
        LayoutOrder            = #self.Active,
    }, self.Container)
    Util.Corner(notif, 12)
    Util.Stroke(notif, accent, 1, 0.6)

    local bar = Util.Create("Frame", {
        Size             = UDim2.new(0, 3, 1, -16),
        Position         = UDim2.new(0, 8, 0, 8),
        BackgroundColor3 = accent,
        ZIndex           = 201,
    }, notif)
    Util.Corner(bar, 3)

    local inner = Util.Create("Frame", {
        Size                   = UDim2.new(1, -24, 0, 0),
        Position               = UDim2.new(0, 20, 0, 0),
        AutomaticSize          = Enum.AutomaticSize.Y,
        BackgroundTransparency = 1,
        ZIndex                 = 201,
    }, notif)
    Util.Pad(inner, 10, 10, 10, 4)
    Util.List(inner, Enum.FillDirection.Vertical, 4)

    Util.Create("TextLabel", {
        Size                   = UDim2.new(1, 0, 0, 18),
        BackgroundTransparency = 1,
        Text                   = title,
        TextColor3             = theme.Text,
        Font                   = Enum.Font.GothamBold,
        TextSize               = Responsive:FontSize(13),
        TextXAlignment         = Enum.TextXAlignment.Left,
        ZIndex                 = 202,
    }, inner)

    if content ~= "" then
        Util.Create("TextLabel", {
            Size                   = UDim2.new(1, 0, 0, 0),
            AutomaticSize          = Enum.AutomaticSize.Y,
            BackgroundTransparency = 1,
            Text                   = content,
            TextColor3             = theme.TextSub,
            Font                   = Enum.Font.Gotham,
            TextSize               = Responsive:FontSize(12),
            TextXAlignment         = Enum.TextXAlignment.Left,
            TextWrapped            = true,
            ZIndex                 = 202,
        }, inner)
    end

    notif.Position = UDim2.new(1, 20, notif.Position.Y.Scale, notif.Position.Y.Offset)
    table.insert(self.Active, notif)
    Util.Tween(notif, Util.TI(0.35, Enum.EasingStyle.Quint), {
        Position = UDim2.new(0, 0, notif.Position.Y.Scale, notif.Position.Y.Offset)
    })

    task.delay(duration, function()
        if notif and notif.Parent then
            Util.Tween(notif, Util.TI(0.3), {
                Position               = UDim2.new(1, 20, notif.Position.Y.Scale, notif.Position.Y.Offset),
                BackgroundTransparency = 1,
            }).Completed:Connect(function()
                pcall(function() notif:Destroy() end)
                for i, v in ipairs(self.Active) do
                    if v == notif then table.remove(self.Active, i); break end
                end
                if #self.Queue > 0 then
                    self:Push(table.remove(self.Queue, 1))
                end
            end)
        end
    end)
end

local StartupPopup = {}

function StartupPopup:Show(screenGui, config)
    config = config or {}
    if not config.Enabled then return end

    local theme    = Themes:Current()
    local title    = config.Title       or "Mixusfield"
    local owner    = config.OwnedBy     or ""
    local desc     = config.Description or ""
    local version  = config.Version     or "1.0.0"
    local duration = config.Duration    or 4
    local icon     = config.Icon        or { Type = "Emoji", Source = "+" }

    local vp       = Util.GetViewport()
    local cardW    = math.min(vp.X - 32, 380)
    local cardH    = 260

    local overlay = Util.Create("Frame", {
        Name                   = "StartupOverlay",
        Size                   = UDim2.new(1, 0, 1, 0),
        BackgroundColor3       = Color3.fromRGB(0, 0, 0),
        BackgroundTransparency = 0.3,
        ZIndex                 = 500,
    }, screenGui)

    local card = Util.Create("Frame", {
        Name                   = "StartupCard",
        Size                   = UDim2.new(0, cardW, 0, cardH),
        Position               = UDim2.new(0.5, -cardW/2, 0.5, -cardH/2),
        BackgroundColor3       = theme.Surface,
        BackgroundTransparency = 0.05,
        ZIndex                 = 501,
    }, screenGui)
    Util.Corner(card, 20)
    Util.Stroke(card, theme.Accent, 1.5, 0.4)
    Util.Pad(card, 28, 28, 28, 28)
    Util.List(card, Enum.FillDirection.Vertical, 10, Enum.HorizontalAlignment.Center)

    -- Icon
    local iconContainer, _ = Icons:Create(card, icon, 48, theme.Accent)
    iconContainer.LayoutOrder = 0

    -- Title
    Util.Create("TextLabel", {
        Size                   = UDim2.new(1, 0, 0, 26),
        BackgroundTransparency = 1,
        Text                   = title,
        TextColor3             = theme.Text,
        Font                   = Enum.Font.GothamBold,
        TextSize               = Responsive:FontSize(20),
        TextXAlignment         = Enum.TextXAlignment.Center,
        ZIndex                 = 502,
        LayoutOrder            = 1,
    }, card)

    if owner ~= "" then
        Util.Create("TextLabel", {
            Size                   = UDim2.new(1, 0, 0, 14),
            BackgroundTransparency = 1,
            Text                   = "by " .. owner .. "  v" .. version,
            TextColor3             = theme.TextSub,
            Font                   = Enum.Font.Gotham,
            TextSize               = Responsive:FontSize(11),
            TextXAlignment         = Enum.TextXAlignment.Center,
            ZIndex                 = 502,
            LayoutOrder            = 2,
        }, card)
    end

    if desc ~= "" then
        Util.Create("TextLabel", {
            Size                   = UDim2.new(1, 0, 0, 0),
            AutomaticSize          = Enum.AutomaticSize.Y,
            BackgroundTransparency = 1,
            Text                   = desc,
            TextColor3             = theme.TextMuted,
            Font                   = Enum.Font.Gotham,
            TextSize               = Responsive:FontSize(12),
            TextXAlignment         = Enum.TextXAlignment.Center,
            TextWrapped            = true,
            ZIndex                 = 502,
            LayoutOrder            = 3,
        }, card)
    end

    -- Progress bar
    local barTrack = Util.Create("Frame", {
        Size             = UDim2.new(1, 0, 0, 4),
        BackgroundColor3 = theme.SurfaceAlt,
        ZIndex           = 502,
        LayoutOrder      = 4,
    }, card)
    Util.Corner(barTrack, 4)
    local barFill = Util.Create("Frame", {
        Size             = UDim2.new(0, 0, 1, 0),
        BackgroundColor3 = theme.Accent,
        ZIndex           = 503,
    }, barTrack)
    Util.Corner(barFill, 4)

    card.BackgroundTransparency  = 1
    overlay.BackgroundTransparency = 1
    Util.Tween(overlay, Util.TI(0.4), { BackgroundTransparency = 0.45 })
    Util.Tween(card,    Util.TI(0.5, Enum.EasingStyle.Back), { BackgroundTransparency = 0.05 })
    Util.Tween(barFill, TweenInfo.new(math.max(duration - 0.5, 0.1), Enum.EasingStyle.Linear), { Size = UDim2.new(1, 0, 1, 0) })

    task.delay(duration, function()
        Util.Tween(card, Util.TI(0.4, Enum.EasingStyle.Quint), {
            BackgroundTransparency = 1,
            Position               = UDim2.new(0.5, -cardW/2, 0.4, -cardH/2),
        }).Completed:Connect(function() pcall(function() card:Destroy() end) end)
        Util.Tween(overlay, Util.TI(0.4), { BackgroundTransparency = 1 })
            .Completed:Connect(function() pcall(function() overlay:Destroy() end) end)
    end)
end

local KeySystem = {}

function KeySystem:Create(screenGui, config, callback, onSuccess)
    config    = config    or {}
    callback  = callback  or function() return true end
    onSuccess = onSuccess or function() end

    local theme    = Themes:Current()
    local title    = config.Title    or "Key System"
    local gateways = config.Gateways or {}
    local note     = config.Note     or "Enter your key to continue."
    local savedKey = config.SaveKey and SaveSystem:Get("MixusfieldKey", "") or ""

    -- Skip UI if saved key is still valid
    if savedKey ~= "" then
        local valid = false
        pcall(function() valid = callback(savedKey) ~= false end)
        if valid then
            onSuccess()
            return
        end
    end

    local kwW, kwH = Responsive:KeySystemSize()

    local overlay = Util.Create("Frame", {
        Name                   = "KeyOverlay",
        Size                   = UDim2.new(1, 0, 1, 0),
        BackgroundColor3       = Color3.fromRGB(0, 0, 0),
        BackgroundTransparency = 1,
        ZIndex                 = 400,
    }, screenGui)
    Util.Tween(overlay, Util.TI(0.3), { BackgroundTransparency = 0.5 })

    -- FIX: Use scale-based centering so it works on all screen sizes
    local win = Util.Create("Frame", {
        Name                   = "KeyWindow",
        Size                   = UDim2.new(0, kwW, 0, kwH),
        Position               = UDim2.new(0.5, -kwW/2, 0.65, -kwH/2),
        BackgroundColor3       = theme.Surface,
        BackgroundTransparency = 1,
        ZIndex                 = 401,
        ClipsDescendants       = true,
    }, screenGui)
    Util.Corner(win, 16)
    Util.Stroke(win, theme.Border, 1, 0.5)

    -- Topbar
    local topH   = Responsive:Scale(44)
    local topbar = Util.Create("Frame", {
        Size             = UDim2.new(1, 0, 0, topH),
        BackgroundColor3 = theme.TopbarBg,
        ZIndex           = 402,
        ClipsDescendants = true,
    }, win)
    Util.Corner(topbar, 16)
    -- Fill bottom half so top corners look square
    Util.Create("Frame", {
        Size             = UDim2.new(1, 0, 0.5, 0),
        Position         = UDim2.new(0, 0, 0.5, 0),
        BackgroundColor3 = theme.TopbarBg,
        ZIndex           = 402,
    }, topbar)

    -- Key icon in topbar
    local keyIconC, _ = Icons:Create(topbar, { Type = "Lucide", Source = "key" }, 18, theme.Accent)
    keyIconC.Position  = UDim2.new(0, 12, 0.5, -9)

    Util.Create("TextLabel", {
        Size                   = UDim2.new(1, -80, 1, 0),
        Position               = UDim2.new(0, 38, 0, 0),
        BackgroundTransparency = 1,
        Text                   = title,
        TextColor3             = theme.Text,
        Font                   = Enum.Font.GothamBold,
        TextSize               = Responsive:FontSize(14),
        TextXAlignment         = Enum.TextXAlignment.Left,
        ZIndex                 = 403,
    }, topbar)

    local closeBtn = Util.Create("TextButton", {
        Size             = UDim2.new(0, 28, 0, 28),
        Position         = UDim2.new(1, -38, 0.5, -14),
        BackgroundColor3 = Color3.fromRGB(200, 60, 60),
        Text             = "X",
        TextColor3       = Color3.fromRGB(255,255,255),
        Font             = Enum.Font.GothamBold,
        TextSize         = Responsive:FontSize(12),
        ZIndex           = 404,
    }, topbar)
    Util.Corner(closeBtn, 8)
    closeBtn.MouseButton1Click:Connect(function()
        Util.Tween(win,     Util.TI(0.25), { BackgroundTransparency = 1 })
            .Completed:Connect(function() pcall(function() win:Destroy() end) end)
        Util.Tween(overlay, Util.TI(0.25), { BackgroundTransparency = 1 })
            .Completed:Connect(function() pcall(function() overlay:Destroy() end) end)

        Util.Tween(closeBtn, Util.TI(0.15), { BackgroundColor3 = Color3.fromRGB(255,80,80) })

        Util.Tween(closeBtn, Util.TI(0.15), { BackgroundColor3 = Color3.fromRGB(200,60,60) })

        Size                 = UDim2.new(1, 0, 1, -topH),

        ScrollBarThickness   = 3,

        Size                   = UDim2.new(1, 0, 0, 0),

        Text                   = note,

    local keyInput = Util.Create("TextBox", {
        Size                   = UDim2.new(1, -16, 1, 0),
        Position               = UDim2.new(0, 8, 0, 0),

        PlaceholderText        = "Enter your key here...",
        PlaceholderColor3      = theme.TextMuted,
        Text                   = "",

        Font                   = Enum.Font.Gotham,
        TextSize               = Responsive:FontSize(13),

        ZIndex                 = 404,

    keyInput.Focused:Connect(function()
        Util.Tween(inputFrame,  Util.TI(0.2), { BackgroundColor3 = theme.Surface })
        Util.Tween(inputStroke, Util.TI(0.2), { Color = theme.Accent, Transparency = 0.2, Thickness = 1.5 })

        Util.Tween(inputFrame,  Util.TI(0.2), { BackgroundColor3 = theme.SurfaceAlt })

    -- Status label

        Size                   = UDim2.new(1, 0, 0, 16),
        BackgroundTransparency = 1,
        Text                   = "",
        TextColor3             = theme.Danger,
        Font                   = Enum.Font.GothamSemibold,
        TextSize               = Responsive:FontSize(11),
        TextXAlignment         = Enum.TextXAlignment.Left,
        ZIndex                 = 403,
        LayoutOrder            = #gateways + 2,
    }, body)

    -- Button row
    local btnRow = Util.Create("Frame", {
        Size                   = UDim2.new(1, 0, 0, Responsive:TapHeight()),
        BackgroundTransparency = 1,
        ZIndex                 = 403,
        LayoutOrder            = #gateways + 3,
    }, body)
    Util.List(btnRow, Enum.FillDirection.Horizontal, 8)

        local btn = Util.Create("TextButton", {
            Size             = UDim2.new(0.5, -4, 1, 0),
            BackgroundColor3 = color,
            Text             = text,
            TextColor3       = Color3.fromRGB(255,255,255),
            Font             = Enum.Font.GothamSemibold,
            TextSize         = Responsive:FontSize(13),
            ZIndex           = 404,
            LayoutOrder      = lo,
        }, btnRow)
        Util.Corner(btn, 8)
        btn.MouseButton1Click:Connect(fn)
        btn.MouseEnter:Connect(function()
            Util.Tween(btn, Util.TI(0.15), { BackgroundColor3 = Util.LerpColor(color, Color3.fromRGB(255,255,255), 0.14) })
        end)
        btn.MouseLeave:Connect(function()
            Util.Tween(btn, Util.TI(0.15), { BackgroundColor3 = color })
        end)
        return btn
    end

    makeKeyBtn("Paste", theme.SurfaceAlt, 1, function()
        if getclipboard then
            local clip = getclipboard()
            if clip and clip ~= "" then keyInput.Text = clip end
        end
    end)

    makeKeyBtn("Submit", theme.Accent, 2, function()
        if submitting then return end
        local key = keyInput.Text
        if key == "" then
            statusLbl.Text      = "Please enter a key first."
            statusLbl.TextColor3 = theme.Warning
            return
        end
        submitting           = true
        statusLbl.Text       = "Verifying..."
        statusLbl.TextColor3 = theme.Info

        task.delay(1.2, function()
            submitting = false
            local valid = false
            pcall(function() valid = callback(key) ~= false end)
            if valid then
                if config.SaveKey then SaveSystem:Set("MixusfieldKey", key) end
                -- FIX: Dismiss key UI first, THEN call onSuccess to reveal the main window
                Util.Tween(win, Util.TI(0.3, Enum.EasingStyle.Quint), {
                    BackgroundTransparency = 1,
                }).Completed:Connect(function()
                    pcall(function() win:Destroy() end)
                end)
                Util.Tween(overlay, Util.TI(0.3), { BackgroundTransparency = 1 })
                    .Completed:Connect(function()
                        pcall(function() overlay:Destroy() end)
                    end)
                task.delay(0.15, function()
                    NotifSystem:Push({ Title = "Key Verified", Content = "Access granted!", Type = "success" })
                    onSuccess() -- <- reveals main window
                end)
            else
                statusLbl.Text       = "Invalid key. Please try again."
                statusLbl.TextColor3 = theme.Danger
                -- Shake animation
                local origPos = inputFrame.Position
                for i = 1, 3 do
                    task.delay(0.06 * (i-1), function()
                        Util.Tween(inputFrame, Util.TI(0.05), {
                            Position = UDim2.new(origPos.X.Scale, origPos.X.Offset + 7,
                                                 origPos.Y.Scale, origPos.Y.Offset)
                        })
                    end)
                    task.delay(0.06 * (i-1) + 0.05, function()
                        Util.Tween(inputFrame, Util.TI(0.05), {
                            Position = UDim2.new(origPos.X.Scale, origPos.X.Offset - 7,
                                                 origPos.Y.Scale, origPos.Y.Offset)
                        })
                    end)
                end
                task.delay(0.3, function() inputFrame.Position = origPos end)
                NotifSystem:Push({ Title = "Invalid Key", Content = "The key you entered is incorrect.", Type = "error" })
            end
        end)
    end)

    -- Animate key window in
    Util.Tween(win, Util.TI(0.45, Enum.EasingStyle.Back), {
        Position               = UDim2.new(0.5, -kwW/2, 0.5, -kwH/2),
        BackgroundTransparency = 0,
    })
end

-- FLOATING BUTTON  (v1.3.0 — larger touch target, clean icon, no gaps)
-- ════════════════════════════════════════════════════════════════════

local FloatingButton = {}

function FloatingButton:Create(screenGui, icon, onClick)
    local theme    = Themes:Current()
    local vp       = Util.GetViewport()
    local btnSz    = Responsive:FabSize()   -- 56 on touch, 48 on desktop
    local corner   = math.floor(btnSz * 0.27)  -- rounded square
    local savedPos = SaveSystem:Get("FloatButtonPos", nil)
    local startPos = savedPos
        and UDim2.new(0, Util.Clamp(savedPos.X, 4, vp.X - btnSz - 4),
                         0, Util.Clamp(savedPos.Y, 4, vp.Y - btnSz - 4))
        or  UDim2.new(0, 10, 0.5, -btnSz/2)  -- left side, vertically centered

    -- Outer glow halo (always behind the button)
    local halo = Util.Create("Frame", {
        Name                   = "FabHalo",
        Size                   = UDim2.new(0, btnSz + 14, 0, btnSz + 14),
        Position               = UDim2.new(0, startPos.X.Offset - 7,
                                            0, startPos.Y.Offset - 7),
        BackgroundColor3       = theme.Accent,
        BackgroundTransparency = 0.78,
        ZIndex                 = 298,
    }, screenGui)
    Util.Corner(halo, corner + 6)

    local btn = Util.Create("Frame", {
        Name             = "FloatingButton",
        Size             = UDim2.new(0, btnSz, 0, btnSz),
        Position         = startPos,
        BackgroundColor3 = theme.Accent,
        ZIndex           = 300,
    }, screenGui)
    Util.Corner(btn, corner)

    -- Subtle inner shine gradient
    Util.Create("UIGradient", {
        Color = ColorSequence.new({
            ColorSequenceKeypoint.new(0, Color3.fromRGB(255,255,255)),
            ColorSequenceKeypoint.new(1, Color3.fromRGB(0,0,0)),
        }),
        Transparency = NumberSequence.new({
            NumberSequenceKeypoint.new(0, 0.82),
            NumberSequenceKeypoint.new(1, 1),
        }),
        Rotation = 135,
    }, btn)

    -- Icon — use a clean Lucide/Symbol icon, no unicode boxes
    local iconSz    = math.floor(btnSz * 0.46)
    local iconC, _  = Icons:Create(btn,
        icon or { Type = "Symbol", Source = "menu" },
        iconSz,
        Color3.fromRGB(255, 255, 255)
    )
    iconC.Position   = UDim2.new(0.5, -iconSz/2, 0.5, -iconSz/2)
    iconC.AnchorPoint = Vector2.new(0, 0)
    iconC.ZIndex     = 302

    local dragging   = false
    local dragStart  = Vector2.new()
    local startFrame = Vector2.new()
    local hasDragged = false

    local clickBtn = Util.Create("TextButton", {
        Size                   = UDim2.new(1, 0, 1, 0),
        BackgroundTransparency = 1,
        Text                   = "",
        ZIndex                 = 305,
    }, btn)

    -- Keep halo synced to button position
    local function syncHalo()
        local bp = btn.AbsolutePosition
        halo.Position = UDim2.new(0, bp.X - 7, 0, bp.Y - 7)
    end

    clickBtn.MouseButton1Down:Connect(function()
        dragging    = true
        hasDragged  = false
        local ip    = UserInputService:GetMouseLocation()
        dragStart   = Vector2.new(ip.X, ip.Y)
        startFrame  = Vector2.new(btn.AbsolutePosition.X, btn.AbsolutePosition.Y)
        -- Press scale
        Util.Tween(btn, Util.TI(0.1, Enum.EasingStyle.Quad), {
            BackgroundColor3 = theme.AccentActive,
        })
    end)

    UserInputService.InputChanged:Connect(function(input)
        if not dragging then return end
        if input.UserInputType == Enum.UserInputType.MouseMovement
        or input.UserInputType == Enum.UserInputType.Touch then
            local delta = Vector2.new(input.Position.X, input.Position.Y) - dragStart
            if delta.Magnitude > 5 then hasDragged = true end
            local nx = Util.Clamp(startFrame.X + delta.X, 4, vp.X - btnSz - 4)
            local ny = Util.Clamp(startFrame.Y + delta.Y, 4, vp.Y - btnSz - 4)
            btn.Position  = UDim2.new(0, nx, 0, ny)
            syncHalo()
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType ~= Enum.UserInputType.MouseButton1
        and input.UserInputType ~= Enum.UserInputType.Touch then return end
        if not dragging then return end
        dragging = false
        Util.Tween(btn, Util.TI(0.15), { BackgroundColor3 = theme.Accent })
        if hasDragged then
            local ap = btn.AbsolutePosition
            local cx = ap.X + btnSz / 2
            local targetX
            if cx < vp.X / 2 then
                targetX = 10
            else
                targetX = vp.X - btnSz - 10
            end
            Util.Tween(btn, Util.TI(0.35, Enum.EasingStyle.Back),
                { Position = UDim2.new(0, targetX, 0, ap.Y) })
            task.delay(0.4, function()
                syncHalo()
                local ap2 = btn.AbsolutePosition
                SaveSystem:Set("FloatButtonPos", { X = ap2.X, Y = ap2.Y })
            end)
        else
            -- Tap: ripple + callback
            Anim.Ripple(btn, btnSz/2, btnSz/2, Color3.fromRGB(255,255,255))
            Util.Tween(halo, Util.TI(0.12), { BackgroundTransparency = 0.45 })
            task.delay(0.35, function()
                Util.Tween(halo, Util.TI(0.3), { BackgroundTransparency = 0.78 })
            end)
            if onClick then Util.Safe(onClick) end
        end
    end)

    clickBtn.MouseEnter:Connect(function()
        Util.Tween(btn,  Util.TI(0.16), { BackgroundColor3 = theme.AccentHover })
        Util.Tween(halo, Util.TI(0.16), { BackgroundTransparency = 0.62 })
    end)
    clickBtn.MouseLeave:Connect(function()
        Util.Tween(btn,  Util.TI(0.16), { BackgroundColor3 = theme.Accent })
        Util.Tween(halo, Util.TI(0.16), { BackgroundTransparency = 0.78 })
    end)

    -- Expose a way to update halo when btn is reparented / repositioned
    btn._halo = halo
    return btn
end

local Components = {}

-- ─── BUTTON ─────────────────────────────────────────────────────────
function Components.Button(parent, config, layoutOrder)
    config = config or {}
    local theme    = Themes:Current()
    local text     = config.Text     or "Button"
    local icon     = config.Icon
    local callback = config.Callback or function() end
    local disabled = config.Disabled or false
    local accent   = config.Accent   or theme.Accent
    local style    = config.Style    or "Filled"

    local h  = Responsive:TapHeight()   -- min 44px on mobile
    local fs = Responsive:FontSize(13)

    local bg, textColor, strokeColor
    if style == "Filled" then
        bg = disabled and theme.SurfaceAlt or accent
        textColor   = Color3.fromRGB(255,255,255)
        strokeColor = nil
    elseif style == "Outline" then
        bg          = Color3.fromRGB(0,0,0)
        textColor   = accent
        strokeColor = accent
    else
        bg          = Color3.fromRGB(0,0,0)
        textColor   = theme.Text
        strokeColor = theme.Border
    end

    local btnFrame = Util.Create("Frame", {
        Size                   = UDim2.new(1, 0, 0, h),
        BackgroundColor3       = bg,
        BackgroundTransparency = style == "Filled" and 0 or 1,
        ZIndex                 = 10,
        LayoutOrder            = layoutOrder or 0,
        ClipsDescendants       = true,
    }, parent)
    Util.Corner(btnFrame, 8)
    if strokeColor then Util.Stroke(btnFrame, strokeColor, 1, 0.5) end

    local inner = Util.Create("Frame", {
        Size                   = UDim2.new(1, 0, 1, 0),
        BackgroundTransparency = 1,
        ZIndex                 = 11,
    }, btnFrame)
    Util.List(inner, Enum.FillDirection.Horizontal, 6, Enum.HorizontalAlignment.Center, Enum.VerticalAlignment.Center)
    Util.Pad(inner, 0, 12, 0, 12)

    if icon then
        local ic, _ = Icons:Create(inner, icon, Responsive:IconSize() - 4, textColor)
        ic.LayoutOrder = 0
    end

    local lbl = Util.Create("TextLabel", {
        Size                   = UDim2.new(0, 0, 1, 0),
        AutomaticSize          = Enum.AutomaticSize.X,
        BackgroundTransparency = 1,
        Text                   = text,
        TextColor3             = textColor,
        Font                   = Enum.Font.GothamSemibold,
        TextSize               = fs,
        ZIndex                 = 12,
        LayoutOrder            = 1,
    }, inner)

    local clickBtn = Util.Create("TextButton", {
        Size                   = UDim2.new(1, 0, 1, 0),
        BackgroundTransparency = 1,
        Text                   = "",
        ZIndex                 = 15,
        Active                 = not disabled,
    }, btnFrame)

    if not disabled then
        clickBtn.MouseButton1Click:Connect(function(x, y)
            local lx = x - btnFrame.AbsolutePosition.X
            local ly = y - btnFrame.AbsolutePosition.Y
            Anim.Ripple(btnFrame, lx, ly, style == "Filled" and Color3.fromRGB(255,255,255) or accent)
            Util.Tween(btnFrame, Util.TI(0.08), { BackgroundColor3 =
                style == "Filled" and Util.LerpColor(accent, Color3.fromRGB(0,0,0), 0.15) or theme.SurfaceAlt
            })
            task.delay(0.12, function()
                Util.Tween(btnFrame, Util.TI(0.15), { BackgroundColor3 = bg })
            end)
            Util.Safe(callback)
        end)
        clickBtn.MouseEnter:Connect(function()
            Util.Tween(btnFrame, Util.TI(0.15), {
                BackgroundColor3 = style == "Filled"
                    and Util.LerpColor(accent, Color3.fromRGB(255,255,255), 0.1)
                    or theme.SurfaceAlt
            })
        end)
        clickBtn.MouseLeave:Connect(function()
            Util.Tween(btnFrame, Util.TI(0.15), { BackgroundColor3 = bg })
        end)
    else
        btnFrame.BackgroundTransparency = 0.4
    end

    local api = {}
    function api:SetText(t) lbl.Text = t end
    function api:SetDisabled(v)
        clickBtn.Active = not v
        btnFrame.BackgroundTransparency = v and 0.4 or 0
    end
    return api
end

-- ─── SLIDER ─────────────────────────────────────────────────────────
function Components.Slider(parent, config, layoutOrder)
    config = config or {}
    local theme    = Themes:Current()
    local text     = config.Text     or "Slider"
    local min      = config.Min      or 0
    local max      = config.Max      or 100
    local step     = config.Step     or 1
    local default  = config.Default  or min
    local callback = config.Callback or function() end
    local saveKey  = config.SaveKey
    local value    = saveKey and SaveSystem:Get(saveKey, default) or default
    value = Util.Clamp(value, min, max)

    local fs = Responsive:FontSize(13)

    local container = Util.Create("Frame", {
        Size             = UDim2.new(1, 0, 0, 0),
        AutomaticSize    = Enum.AutomaticSize.Y,
        BackgroundColor3 = theme.SurfaceAlt,
        ZIndex           = 10,
        LayoutOrder      = layoutOrder or 0,
    }, parent)
    Util.Corner(container, 8)
    Util.Pad(container, 10, 12, 14, 12)
    Util.List(container, Enum.FillDirection.Vertical, 10)

    local topRow = Util.Create("Frame", {
        Size                   = UDim2.new(1, 0, 0, 18),
        BackgroundTransparency = 1,
        ZIndex                 = 11,
    }, container)

    Util.Create("TextLabel", {
        Size                   = UDim2.new(1, -60, 1, 0),
        BackgroundTransparency = 1,
        Text                   = text,
        TextColor3             = theme.Text,
        Font                   = Enum.Font.Gotham,
        TextSize               = fs,
        TextXAlignment         = Enum.TextXAlignment.Left,
        ZIndex                 = 12,
    }, topRow)

    local valLbl = Util.Create("TextLabel", {
        Size                   = UDim2.new(0, 55, 1, 0),
        Position               = UDim2.new(1, -55, 0, 0),
        BackgroundTransparency = 1,
        Text                   = tostring(value),
        TextColor3             = theme.Accent,
        Font                   = Enum.Font.GothamBold,
        TextSize               = fs,
        TextXAlignment         = Enum.TextXAlignment.Right,
        ZIndex                 = 12,
    }, topRow)

    local trackH  = Responsive:Scale(6)
    local trackBg = Util.Create("Frame", {
        Size             = UDim2.new(1, 0, 0, trackH),
        BackgroundColor3 = theme.Border,
        ZIndex           = 11,
    }, container)
    Util.Corner(trackBg, trackH/2)

    local fill = Util.Create("Frame", {
        Size             = UDim2.new(0, 0, 1, 0),
        BackgroundColor3 = theme.Accent,
        ZIndex           = 12,
    }, trackBg)
    Util.Corner(fill, trackH/2)

    local thumbSz = Responsive:Scale(20)
    local thumb = Util.Create("Frame", {
        Size             = UDim2.new(0, thumbSz, 0, thumbSz),
        Position         = UDim2.new(0, 0, 0.5, -thumbSz/2),
        BackgroundColor3 = Color3.fromRGB(255,255,255),
        ZIndex           = 13,
    }, trackBg)
    Util.Corner(thumb, thumbSz/2)
    Util.Stroke(thumb, theme.Accent, 2, 0.3)

    local function getValue(frac)
        local raw = min + (max - min) * Util.Clamp(frac, 0, 1)
        if step > 0 then raw = Util.Round(raw, step) end
        return Util.Clamp(raw, min, max)
    end

    local function updateVisual(v, animate)
        local frac = (max ~= min) and (v - min) / (max - min) or 0
        local dur  = animate and 0.08 or 0
        Util.Tween(fill,  Util.TI(dur, Enum.EasingStyle.Linear), { Size = UDim2.new(frac, 0, 1, 0) })
        Util.Tween(thumb, Util.TI(dur, Enum.EasingStyle.Linear), {
            Position = UDim2.new(frac, -thumbSz/2, 0.5, -thumbSz/2)
        })
        valLbl.Text = tostring(v)
    end
    updateVisual(value, false)

    local dragging = false

    local function handleDrag(inputX)
        local trackPos  = trackBg.AbsolutePosition.X
        local trackSize = trackBg.AbsoluteSize.X
        if trackSize == 0 then return end
        local frac = (inputX - trackPos) / trackSize
        local v    = getValue(frac)
        if v ~= value then
            value = v
            if saveKey then SaveSystem:Set(saveKey, v) end
            updateVisual(v, true)
            Util.Safe(callback, v)
        end
    end

    local hitBtn = Util.Create("TextButton", {
        Size                   = UDim2.new(1, 0, 0, trackH + thumbSz + 8),
        Position               = UDim2.new(0, 0, 0.5, -(trackH + thumbSz + 8)/2),
        BackgroundTransparency = 1,
        Text                   = "",
        ZIndex                 = 14,
    }, trackBg)

    hitBtn.MouseButton1Down:Connect(function(x) dragging = true; handleDrag(x) end)

    UserInputService.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.Touch then
            local pos = input.Position
            local ap  = hitBtn.AbsolutePosition
            local sz  = hitBtn.AbsoluteSize
            if pos.X >= ap.X and pos.X <= ap.X + sz.X
            and pos.Y >= ap.Y and pos.Y <= ap.Y + sz.Y then
                dragging = true; handleDrag(pos.X)
            end
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if not dragging then return end
        if input.UserInputType == Enum.UserInputType.MouseMovement
        or input.UserInputType == Enum.UserInputType.Touch then
            handleDrag(input.Position.X)
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then
            dragging = false
        end
    end)

    local api = {}
    function api:SetValue(v)
        value = Util.Clamp(v, min, max)
        updateVisual(value, true)
    end
    function api:GetValue() return value end
    return api
end

-- ─── DROPDOWN ───────────────────────────────────────────────────────
function Components.Dropdown(parent, config, layoutOrder)
    config = config or {}
    local theme    = Themes:Current()
    local text     = config.Text     or "Dropdown"
    local options  = config.Options  or {}
    local multi    = config.Multi    or false
    local default  = config.Default  or (multi and {} or nil)
    local callback = config.Callback or function() end
    local saveKey  = config.SaveKey
    local selected = saveKey and SaveSystem:Get(saveKey, default) or default

    local fs   = Responsive:FontSize(13)
    local open = false

    local wrapper = Util.Create("Frame", {
        Size                   = UDim2.new(1, 0, 0, 0),
        AutomaticSize          = Enum.AutomaticSize.Y,
        BackgroundTransparency = 1,
        ZIndex                 = 20,
        LayoutOrder            = layoutOrder or 0,
        ClipsDescendants       = false,
    }, parent)

    local header = Util.Create("Frame", {
        Size             = UDim2.new(1, 0, 0, Responsive:TapHeight()),
        BackgroundColor3 = theme.SurfaceAlt,
        ZIndex           = 21,
    }, wrapper)
    Util.Corner(header, 8)
    Util.Pad(header, 0, 10, 0, 12)

    local headerRow = Util.Create("Frame", {
        Size                   = UDim2.new(1, 0, 1, 0),
        BackgroundTransparency = 1,
        ZIndex                 = 22,
    }, header)

    Util.Create("TextLabel", {
        Size                   = UDim2.new(1, -30, 0.5, 0),
        Position               = UDim2.new(0, 0, 0, 0),
        BackgroundTransparency = 1,
        Text                   = text,
        TextColor3             = theme.TextMuted,
        Font                   = Enum.Font.Gotham,
        TextSize               = Responsive:FontSize(10),
        TextXAlignment         = Enum.TextXAlignment.Left,
        ZIndex                 = 23,
    }, headerRow)

    local function getDisplayText()
        if multi then
            if #selected == 0 then return "None selected"
            elseif #selected == 1 then return selected[1]
            else return tostring(#selected) .. " selected" end
        else return selected or "Select..." end
    end

    local selLbl = Util.Create("TextLabel", {
        Size                   = UDim2.new(1, -30, 0.5, 0),
        Position               = UDim2.new(0, 0, 0.5, 0),
        BackgroundTransparency = 1,
        Text                   = getDisplayText(),
        TextColor3             = theme.Text,
        Font                   = Enum.Font.GothamSemibold,
        TextSize               = fs,
        TextXAlignment         = Enum.TextXAlignment.Left,
        ZIndex                 = 23,
    }, headerRow)

    local arrowLbl = Util.Create("TextLabel", {
        Size                   = UDim2.new(0, 20, 1, 0),
        Position               = UDim2.new(1, -20, 0, 0),
        BackgroundTransparency = 1,
        Text                   = "v",
        TextColor3             = theme.TextSub,
        Font                   = Enum.Font.GothamBold,
        TextSize               = Responsive:FontSize(12),
        ZIndex                 = 23,
    }, headerRow)

    local listFrame = Util.Create("Frame", {
        Size             = UDim2.new(1, 0, 0, 0),
        Position         = UDim2.new(0, 0, 1, 4),
        BackgroundColor3 = theme.Surface,
        ZIndex           = 50,
        ClipsDescendants = true,
        Visible          = false,
    }, header)
    Util.Corner(listFrame, 8)
    Util.Stroke(listFrame, theme.Border, 1, 0.5)

    local listInner = Util.Create("ScrollingFrame", {
        Size                   = UDim2.new(1, 0, 1, 0),
        BackgroundTransparency = 1,
        ScrollBarThickness     = 3,
        ScrollBarImageColor3   = theme.Accent,
        CanvasSize             = UDim2.new(0, 0, 0, 0),
        AutomaticCanvasSize    = Enum.AutomaticSize.Y,
        ZIndex                 = 51,
    }, listFrame)
    Util.Pad(listInner, 6)
    Util.List(listInner, Enum.FillDirection.Vertical, 2)

    local function isSelected(opt)
        if multi then
            for _, v in ipairs(selected) do if v == opt then return true end end
            return false
        else return selected == opt end
    end

    local function buildList()
        for _, child in ipairs(listInner:GetChildren()) do
            if child:IsA("Frame") or child:IsA("TextButton") then child:Destroy() end
        end
        for _, opt in ipairs(options) do
            local isSel = isSelected(opt)
            local item = Util.Create("TextButton", {
                Size                   = UDim2.new(1, 0, 0, Responsive:Scale(34)),
                BackgroundColor3       = isSel and theme.SurfaceAlt or Color3.fromRGB(0,0,0),
                BackgroundTransparency = isSel and 0 or 1,
                Text                   = opt,
                TextColor3             = isSel and theme.Accent or theme.Text,
                Font                   = isSel and Enum.Font.GothamSemibold or Enum.Font.Gotham,
                TextSize               = fs,
                TextXAlignment         = Enum.TextXAlignment.Left,
                ZIndex                 = 52,
            }, listInner)
            Util.Corner(item, 6)
            Util.Pad(item, 0, 8, 0, 8)

            item.MouseButton1Click:Connect(function()
                if multi then
                    if isSelected(opt) then
                        for i, v in ipairs(selected) do
                            if v == opt then table.remove(selected, i); break end
                        end
                    else
                        table.insert(selected, opt)
                    end
                else
                    selected = opt
                    open     = false
                    Anim.SlideClose(listFrame, 0.2, function() listFrame.Visible = false end)
                    Util.Tween(arrowLbl, Util.TI(0.2), { Rotation = 0 })
                end
                if saveKey then SaveSystem:Set(saveKey, selected) end
                selLbl.Text = getDisplayText()
                buildList()
                Util.Safe(callback, selected)
            end)
            item.MouseEnter:Connect(function()
                if not isSel then
                    Util.Tween(item, Util.TI(0.12), { BackgroundTransparency = 0, BackgroundColor3 = theme.SurfaceAlt })
                end
            end)
            item.MouseLeave:Connect(function()
                if not isSelected(opt) then
                    Util.Tween(item, Util.TI(0.12), { BackgroundTransparency = 1 })
                end
            end)
        end
    end
    buildList()

    local headerBtn = Util.Create("TextButton", {
        Size                   = UDim2.new(1, 0, 1, 0),
        BackgroundTransparency = 1,
        Text                   = "",
        ZIndex                 = 24,
    }, header)

    headerBtn.MouseButton1Click:Connect(function()
        open = not open
        if open then
            listFrame.Visible = true
            local itemH = Responsive:Scale(34) + 2
            local maxH  = math.min(#options * itemH + 16, 180)
            Anim.SlideOpen(listFrame, UDim2.new(1, 0, 0, maxH), 0.2)
            Util.Tween(arrowLbl, Util.TI(0.2), { Rotation = 180 })
        else
            Anim.SlideClose(listFrame, 0.2, function() listFrame.Visible = false end)
            Util.Tween(arrowLbl, Util.TI(0.2), { Rotation = 0 })
        end
    end)

    local api = {}
    function api:SetOptions(opts) options = opts; buildList() end
    function api:GetSelected() return selected end
    function api:SetSelected(v) selected = v; selLbl.Text = getDisplayText(); buildList() end
    return api
end

-- ─── KEYBIND ────────────────────────────────────────────────────────
function Components.Keybind(parent, config, layoutOrder)
    config = config or {}
    local theme    = Themes:Current()
    local text     = config.Text     or "Keybind"
    local default  = config.Default  or Enum.KeyCode.Unknown
    local callback = config.Callback or function() end
    local saveKey  = config.SaveKey
    local mode     = config.Mode     or "Toggle"

    local fs         = Responsive:FontSize(13)
    local currentKey = saveKey and Enum.KeyCode[SaveSystem:Get(saveKey, default.Name)] or default
    local listening  = false
    local active     = false
    local holding    = false

    local row = Util.Create("Frame", {
        Size             = UDim2.new(1, 0, 0, Responsive:TapHeight()),
        BackgroundColor3 = theme.SurfaceAlt,
        ZIndex           = 10,
        LayoutOrder      = layoutOrder or 0,
    }, parent)
    Util.Corner(row, 8)
    Util.Pad(row, 0, 12, 0, 12)

    Util.Create("TextLabel", {
        Size                   = UDim2.new(1, -100, 1, 0),
        BackgroundTransparency = 1,
        Text                   = text,
        TextColor3             = theme.Text,
        Font                   = Enum.Font.Gotham,
        TextSize               = fs,
        TextXAlignment         = Enum.TextXAlignment.Left,
        ZIndex                 = 11,
    }, row)

    local keyBadge = Util.Create("TextButton", {
        Size             = UDim2.new(0, 90, 0, 28),
        Position         = UDim2.new(1, -90, 0.5, -14),
        BackgroundColor3 = theme.Surface,
        Text             = currentKey.Name,
        TextColor3       = theme.Accent,
        Font             = Enum.Font.GothamBold,
        TextSize         = Responsive:FontSize(11),
        ZIndex           = 11,
    }, row)
    Util.Corner(keyBadge, 6)
    Util.Stroke(keyBadge, theme.Accent, 1, 0.5)

    keyBadge.MouseButton1Click:Connect(function()
        if listening then return end
        listening       = true
        keyBadge.Text   = "Press..."
        keyBadge.TextColor3 = theme.Warning
        Util.Tween(keyBadge, Util.TI(0.15), { BackgroundColor3 = theme.SurfaceAlt })
    end)

    local inputConn = UserInputService.InputBegan:Connect(function(input, gpe)
        if not listening then
            if input.KeyCode == currentKey then
                if mode == "Toggle" then
                    active = not active
                    Util.Safe(callback, active)
                elseif mode == "Hold" then
                    holding = true; active = true
                    Util.Safe(callback, true)
                end
            end
        else
            if input.UserInputType == Enum.UserInputType.Keyboard then
                currentKey = input.KeyCode
                if saveKey then SaveSystem:Set(saveKey, input.KeyCode.Name) end
                keyBadge.Text       = input.KeyCode.Name
                keyBadge.TextColor3 = theme.Accent
                Util.Tween(keyBadge, Util.TI(0.15), { BackgroundColor3 = theme.Surface })
                listening = false
            end
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if mode == "Hold" and input.KeyCode == currentKey and holding then
            holding = false; active = false
            Util.Safe(callback, false)
        end
    end)

    local api = {}
    function api:SetKey(kc) currentKey = kc; keyBadge.Text = kc.Name end
    function api:GetKey() return currentKey end
    function api:Cleanup() if inputConn then inputConn:Disconnect() end end
    return api
end

-- ─── PROGRESS BAR ───────────────────────────────────────────────────
function Components.ProgressBar(parent, config, layoutOrder)
    config = config or {}
    local theme    = Themes:Current()
    local text     = config.Text    or "Progress"
    local value    = Util.Clamp(config.Value or 0, 0, 100)
    local color    = config.Color   or theme.Accent
    local showPct  = config.ShowPct ~= false

    local container = Util.Create("Frame", {
        Size             = UDim2.new(1, 0, 0, 0),
        AutomaticSize    = Enum.AutomaticSize.Y,
        BackgroundColor3 = theme.SurfaceAlt,
        ZIndex           = 10,
        LayoutOrder      = layoutOrder or 0,
    }, parent)
    Util.Corner(container, 8)
    Util.Pad(container, 10, 12, 12, 12)
    Util.List(container, Enum.FillDirection.Vertical, 6)

    -- Label row
    local topRow = Util.Create("Frame", {
        Size                   = UDim2.new(1, 0, 0, 16),
        BackgroundTransparency = 1,
        ZIndex                 = 11,
    }, container)

    Util.Create("TextLabel", {
        Size                   = UDim2.new(1, -50, 1, 0),
        BackgroundTransparency = 1,
        Text                   = text,
        TextColor3             = theme.Text,
        Font                   = Enum.Font.Gotham,
        TextSize               = Responsive:FontSize(12),
        TextXAlignment         = Enum.TextXAlignment.Left,
        ZIndex                 = 12,
    }, topRow)

    local pctLbl = Util.Create("TextLabel", {
        Size                   = UDim2.new(0, 45, 1, 0),
        Position               = UDim2.new(1, -45, 0, 0),
        BackgroundTransparency = 1,
        Text                   = tostring(value) .. "%",
        TextColor3             = color,
        Font                   = Enum.Font.GothamBold,
        TextSize               = Responsive:FontSize(12),
        TextXAlignment         = Enum.TextXAlignment.Right,
        ZIndex                 = 12,
        Visible                = showPct,
    }, topRow)

    -- Track
    local trackH = Responsive:Scale(8)
    local trackBg = Util.Create("Frame", {
        Size             = UDim2.new(1, 0, 0, trackH),
        BackgroundColor3 = theme.Border,
        ZIndex           = 11,
    }, container)
    Util.Corner(trackBg, trackH / 2)

    local fill = Util.Create("Frame", {
        Size             = UDim2.new(value / 100, 0, 1, 0),
        BackgroundColor3 = color,
        ZIndex           = 12,
    }, trackBg)
    Util.Corner(fill, trackH / 2)

    -- Animated shimmer stripe
    local shimmer = Util.Create("Frame", {
        Size                   = UDim2.new(0, 60, 1, 0),
        BackgroundColor3       = Color3.fromRGB(255, 255, 255),
        BackgroundTransparency = 0.75,
        ZIndex                 = 13,
        ClipsDescendants       = false,
    }, fill)
    Util.Corner(shimmer, trackH / 2)

    -- Animate shimmer continuously
    local function loopShimmer()
        shimmer.Position = UDim2.new(0, -60, 0, 0)
        if fill.AbsoluteSize.X < 10 then return end
        Util.Tween(shimmer, TweenInfo.new(1.2, Enum.EasingStyle.Linear), {
            Position = UDim2.new(1, 60, 0, 0)
        }).Completed:Connect(function()
            task.wait(0.5)
            loopShimmer()
        end)
    end
    task.delay(0.5, loopShimmer)

    local api = {}
    function api:SetValue(v)
        v = Util.Clamp(v, 0, 100)
        value = v
        Util.Tween(fill, Util.TI(0.4, Enum.EasingStyle.Quint), {
            Size = UDim2.new(v / 100, 0, 1, 0)
        })
        pctLbl.Text = tostring(math.round(v)) .. "%"
    end
    function api:GetValue() return value end
    function api:SetColor(c)
        color = c
        fill.BackgroundColor3 = c
        pctLbl.TextColor3     = c
    end
    return api
end

-- ─── COLOR PICKER ───────────────────────────────────────────────────
-- Full HSV color picker with hex input, opacity, and save support
function Components.ColorPicker(parent, config, layoutOrder)
    config = config or {}
    local theme    = Themes:Current()
    local text     = config.Text     or "Color"
    local default  = config.Default  or Color3.fromRGB(120, 100, 255)
    local callback = config.Callback or function() end
    local saveKey  = config.SaveKey
    local opacity  = config.Opacity ~= false  -- show opacity slider

    local savedRaw = saveKey and SaveSystem:Get(saveKey, nil) or nil
    local currentColor
    if savedRaw then
        pcall(function()
            currentColor = Color3.fromRGB(savedRaw.R, savedRaw.G, savedRaw.B)
        end)
    end
    currentColor = currentColor or default

    local h, s, v = Color3.toHSV(currentColor)
    local alpha    = Util.Clamp(config.Alpha or 1, 0, 1)
    local open     = false

    -- Outer wrapper (collapsed by default)
    local wrapper = Util.Create("Frame", {
        Size                   = UDim2.new(1, 0, 0, 0),
        AutomaticSize          = Enum.AutomaticSize.Y,
        BackgroundTransparency = 1,
        ZIndex                 = 20,
        LayoutOrder            = layoutOrder or 0,
        ClipsDescendants       = false,
    }, parent)
    Util.List(wrapper, Enum.FillDirection.Vertical, 6)

    -- Header row (clickable)
    local header = Util.Create("Frame", {
        Size             = UDim2.new(1, 0, 0, Responsive:TapHeight()),
        BackgroundColor3 = theme.SurfaceAlt,
        ZIndex           = 21,
    }, wrapper)
    Util.Corner(header, 8)
    Util.Pad(header, 0, 12, 0, 12)

    Util.Create("TextLabel", {
        Size                   = UDim2.new(1, -80, 1, 0),
        BackgroundTransparency = 1,
        Text                   = text,
        TextColor3             = theme.Text,
        Font                   = Enum.Font.Gotham,
        TextSize               = Responsive:FontSize(13),
        TextXAlignment         = Enum.TextXAlignment.Left,
        ZIndex                 = 22,
    }, header)

    -- Color preview swatch
    local swatch = Util.Create("Frame", {
        Size             = UDim2.new(0, 28, 0, 22),
        Position         = UDim2.new(1, -60, 0.5, -11),
        BackgroundColor3 = currentColor,
        ZIndex           = 22,
    }, header)
    Util.Corner(swatch, 6)
    Util.Stroke(swatch, theme.Border, 1, 0.5)

    local arrowLbl = Util.Create("TextLabel", {
        Size                   = UDim2.new(0, 18, 1, 0),
        Position               = UDim2.new(1, -20, 0, 0),
        BackgroundTransparency = 1,
        Text                   = "v",
        TextColor3             = theme.TextSub,
        Font                   = Enum.Font.GothamBold,
        TextSize               = Responsive:FontSize(12),
        ZIndex                 = 22,
    }, header)

    -- Picker panel (expanded)
    local panel = Util.Create("Frame", {
        Size                   = UDim2.new(1, 0, 0, 0),
        BackgroundColor3       = theme.Surface,
        ZIndex                 = 21,
        Visible                = false,
        ClipsDescendants       = true,
    }, wrapper)
    Util.Corner(panel, 8)
    Util.Stroke(panel, theme.Border, 1, 0.5)
    Util.Pad(panel, 10, 12, 12, 12)
    Util.List(panel, Enum.FillDirection.Vertical, 8)

    -- Helper: make a gradient slider (hue / saturation / value / alpha)
    local function makeGradSlider(gradColors, labelText, initFrac, lo)
        local slotH = Responsive:Scale(20)
        local slotContainer = Util.Create("Frame", {
            Size                   = UDim2.new(1, 0, 0, 0),
            AutomaticSize          = Enum.AutomaticSize.Y,
            BackgroundTransparency = 1,
            ZIndex                 = 22,
            LayoutOrder            = lo or 0,
        }, panel)
        Util.List(slotContainer, Enum.FillDirection.Vertical, 4)

        if labelText then
            Util.Create("TextLabel", {
                Size                   = UDim2.new(1, 0, 0, 13),
                BackgroundTransparency = 1,
                Text                   = labelText,
                TextColor3             = theme.TextMuted,
                Font                   = Enum.Font.Gotham,
                TextSize               = Responsive:FontSize(10),
                TextXAlignment         = Enum.TextXAlignment.Left,
                ZIndex                 = 23,
            }, slotContainer)
        end

        local track = Util.Create("Frame", {
            Size             = UDim2.new(1, 0, 0, slotH),
            BackgroundColor3 = Color3.fromRGB(255,255,255),
            ZIndex           = 22,
        }, slotContainer)
        Util.Corner(track, slotH / 2)
        Util.Create("UIGradient", { Color = gradColors }, track)
        Util.Stroke(track, theme.Border, 1, 0.7)

        local thumbSz = slotH + 2
        local thumb = Util.Create("Frame", {
            Size             = UDim2.new(0, thumbSz, 0, thumbSz),
            Position         = UDim2.new(initFrac, -thumbSz/2, 0.5, -thumbSz/2),
            BackgroundColor3 = Color3.fromRGB(255,255,255),
            ZIndex           = 23,
        }, track)
        Util.Corner(thumb, thumbSz/2)
        Util.Stroke(thumb, Color3.fromRGB(200,200,200), 1.5, 0.2)

        local draggingSlider = false

        local function setFrac(frac)
            frac = Util.Clamp(frac, 0, 1)
            thumb.Position = UDim2.new(frac, -thumbSz/2, 0.5, -thumbSz/2)
            return frac
        end

        local hitBtn = Util.Create("TextButton", {
            Size                   = UDim2.new(1, 0, 0, slotH + thumbSz),
            Position               = UDim2.new(0, 0, 0.5, -(slotH + thumbSz)/2),
            BackgroundTransparency = 1,
            Text                   = "",
            ZIndex                 = 24,
        }, track)

        local onChange = nil  -- caller sets this

        local function drag(inputX)
            local ap = track.AbsolutePosition.X
            local sz = track.AbsoluteSize.X
            if sz == 0 then return end
            local frac = Util.Clamp((inputX - ap) / sz, 0, 1)
            setFrac(frac)
            if onChange then onChange(frac) end
        end

        hitBtn.MouseButton1Down:Connect(function(x) draggingSlider = true; drag(x) end)

        UserInputService.InputChanged:Connect(function(input)
            if not draggingSlider then return end
            if input.UserInputType == Enum.UserInputType.MouseMovement
            or input.UserInputType == Enum.UserInputType.Touch then
                drag(input.Position.X)
            end
        end)

        UserInputService.InputEnded:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1
            or input.UserInputType == Enum.UserInputType.Touch then
                draggingSlider = false
            end
        end)

        UserInputService.InputBegan:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.Touch then
                local pos = input.Position
                local ap  = hitBtn.AbsolutePosition
                local sz  = hitBtn.AbsoluteSize
                if pos.X >= ap.X and pos.X <= ap.X + sz.X
                and pos.Y >= ap.Y and pos.Y <= ap.Y + sz.Y then
                    draggingSlider = true; drag(pos.X)
                end
            end
        end)

        return setFrac, function(fn) onChange = fn end
    end

    -- Build the hue gradient (full rainbow)
    local hueKeys = {}
    for i = 0, 6 do
        hueKeys[i+1] = ColorSequenceKeypoint.new(
            i / 6, Color3.fromHSV(i / 6, 1, 1)
        )
    end
    local hueGrad = ColorSequence.new(hueKeys)

    -- Helper: get SV gradient for current hue
    local function svGrad()
        return ColorSequence.new({
            ColorSequenceKeypoint.new(0, Color3.fromHSV(h, 0, 1)),
            ColorSequenceKeypoint.new(1, Color3.fromHSV(h, 1, 1)),
        })
    end
    local function valGrad()
        return ColorSequence.new({
            ColorSequenceKeypoint.new(0, Color3.fromRGB(0,0,0)),
            ColorSequenceKeypoint.new(1, Color3.fromHSV(h, s, 1)),
        })
    end

    -- Current color preview inside panel
    local previewRow = Util.Create("Frame", {
        Size                   = UDim2.new(1, 0, 0, Responsive:Scale(36)),
        BackgroundTransparency = 1,
        ZIndex                 = 22,
        LayoutOrder            = 0,
    }, panel)

    local bigSwatch = Util.Create("Frame", {
        Size             = UDim2.new(1, -60, 1, 0),
        BackgroundColor3 = currentColor,
        ZIndex           = 23,
    }, previewRow)
    Util.Corner(bigSwatch, 8)
    Util.Stroke(bigSwatch, theme.Border, 1, 0.5)

    -- Hex label
    local hexLbl = Util.Create("TextLabel", {
        Size                   = UDim2.new(0, 54, 1, 0),
        Position               = UDim2.new(1, -54, 0, 0),
        BackgroundTransparency = 1,
        Text                   = "#" .. string.format("%02X%02X%02X",
            math.round(currentColor.R * 255),
            math.round(currentColor.G * 255),
            math.round(currentColor.B * 255)),
        TextColor3             = theme.TextSub,
        Font                   = Enum.Font.GothamBold,
        TextSize               = Responsive:FontSize(11),
        TextXAlignment         = Enum.TextXAlignment.Right,
        ZIndex                 = 23,
    }, previewRow)

    local function updateColor()
        currentColor = Color3.fromHSV(h, s, v)
        swatch.BackgroundColor3   = currentColor
        bigSwatch.BackgroundColor3 = currentColor
        hexLbl.Text = "#" .. string.format("%02X%02X%02X",
            math.round(currentColor.R * 255),
            math.round(currentColor.G * 255),
            math.round(currentColor.B * 255))
        if saveKey then
            SaveSystem:Set(saveKey, {
                R = math.round(currentColor.R * 255),
                G = math.round(currentColor.G * 255),
                B = math.round(currentColor.B * 255),
            })
        end
        Util.Safe(callback, currentColor, alpha)
    end

    -- Hue slider
    local setHueFrac, onHueChange = makeGradSlider(hueGrad, "Hue", h, 1)
    onHueChange(function(frac)
        h = frac
        updateColor()
    end)

    -- Saturation slider (gradient updates with hue)
    local satGradInst
    local setSatFrac, onSatChange = makeGradSlider(svGrad(), "Saturation", s, 2)
    onSatChange(function(frac)
        s = frac
        updateColor()
    end)

    -- Value/Brightness slider
    local setBriGrad
    local setValFrac, onValChange = makeGradSlider(valGrad(), "Brightness", v, 3)
    onValChange(function(frac)
        v = frac
        updateColor()
    end)

    -- Alpha slider (optional)
    if opacity then
        local alphaGrad = ColorSequence.new({
            ColorSequenceKeypoint.new(0, Color3.fromRGB(0,0,0)),
            ColorSequenceKeypoint.new(1, Color3.fromRGB(255,255,255)),
        })
        local setAlphaFrac, onAlphaChange = makeGradSlider(alphaGrad, "Opacity", alpha, 4)
        onAlphaChange(function(frac)
            alpha = frac
            Util.Safe(callback, currentColor, alpha)
        end)
    end

    -- Toggle open/close
    local headerBtn = Util.Create("TextButton", {
        Size                   = UDim2.new(1, 0, 1, 0),
        BackgroundTransparency = 1,
        Text                   = "",
        ZIndex                 = 25,
    }, header)

    local panelH = opacity and 260 or 215
    headerBtn.MouseButton1Click:Connect(function()
        open = not open
        Util.Tween(arrowLbl, Util.TI(0.2), { Rotation = open and 180 or 0 })
        if open then
            panel.Visible = true
            Anim.SlideOpen(panel, UDim2.new(1, 0, 0, Responsive:Scale(panelH)), 0.25)
        else
            Anim.SlideClose(panel, 0.2, function() panel.Visible = false end)
        end
    end)

    local api = {}
    function api:GetColor()  return currentColor end
    function api:GetAlpha()  return alpha end
    function api:SetColor(c)
        currentColor = c
        h, s, v = Color3.toHSV(c)
        setHueFrac(h); setSatFrac(s); setValFrac(v)
        swatch.BackgroundColor3    = c
        bigSwatch.BackgroundColor3 = c
        hexLbl.Text = "#" .. string.format("%02X%02X%02X",
            math.round(c.R*255), math.round(c.G*255), math.round(c.B*255))
    end
    return api
end

        ZIndex                 = 11,

    Util.Pad(self.Header, 0, 0, 0, 2)

    local titleOffset = 0
    if self.Icon then
        local ic, _ = Icons:Create(self.Header, self.Icon, 16, theme.TextMuted)
        ic.Position   = UDim2.new(0, 0, 0.5, -8)

        ZIndex                 = 12,

    if self.Collapsible then
        self.Arrow = Util.Create("TextLabel", {
            Size                   = UDim2.new(0, 16, 1, 0),
            Position               = UDim2.new(1, -16, 0, 0),
            BackgroundTransparency = 1,
            Text                   = "v",
            TextColor3             = theme.TextMuted,
            Font                   = Enum.Font.GothamBold,
            TextSize               = Responsive:FontSize(11),
            ZIndex                 = 12,
        }, self.Header)

        local hBtn = Util.Create("TextButton", {
            Size                   = UDim2.new(1, 0, 1, 0),
            BackgroundTransparency = 1,
            Text                   = "",
            ZIndex                 = 13,
        }, self.Header)
        hBtn.MouseButton1Click:Connect(function() self:ToggleCollapse() end)
    end

    self.Content = Util.Create("Frame", {
        Size                   = UDim2.new(1, 0, 0, 0),
        AutomaticSize          = Enum.AutomaticSize.Y,
        BackgroundTransparency = 1,
        ZIndex                 = 11,
        ClipsDescendants       = false,
    }, self.Frame)
    Util.Pad(self.Content, 0, 0, 4, 2)
    Util.List(self.Content, Enum.FillDirection.Vertical, 4)

    if self.Collapsed then self:ToggleCollapse(true) end
    return self
end

function Section:ToggleCollapse(instant)
    self.Collapsed = not self.Collapsed
    if self.Arrow then
        Util.Tween(self.Arrow, Util.TI(instant and 0 or 0.2), {
            Rotation = self.Collapsed and -90 or 0
        })
    end
    self.Content.Visible = not self.Collapsed
end

-- SEARCH ENGINE
-- ════════════════════════════════════════════════════════════════════

local SearchEngine = {}

function SearchEngine:Init(searchBox, tabs)
    if not searchBox then return end
    searchBox:GetPropertyChangedSignal("Text"):Connect(function()
        local query = string.lower(searchBox.Text)
        for _, tab in ipairs(tabs) do
            for _, sec in ipairs(tab.Sections or {}) do
                if query == "" then
                    sec.Frame.Visible = true
                    for _, child in ipairs(sec.Content:GetChildren()) do
                        if child:IsA("GuiObject") and not child:IsA("UIListLayout") and not child:IsA("UIPadding") then
                            child.Visible = true
                        end
                    end
                else
                    local secMatch = string.find(string.lower(sec.Title), query, 1, true)
                    local anyVis   = secMatch ~= nil
                    for _, child in ipairs(sec.Content:GetChildren()) do
                        if child:IsA("GuiObject") and not child:IsA("UIListLayout") and not child:IsA("UIPadding") then
                            local lbl   = child:FindFirstChildWhichIsA("TextLabel")
                            local match = lbl and string.find(string.lower(lbl.Text or ""), query, 1, true)
                            child.Visible = (match ~= nil) or (secMatch ~= nil)
                            if child.Visible then anyVis = true end
                        end
                    end
                    sec.Frame.Visible = anyVis
                end
            end
        end
    end)
end

function Window:AddTab(config)
    local tab = Tab.new(self.TabBar, self.ContentArea, config)
    table.insert(self.Tabs, tab)

    -- Sidebar icon — only if sidebar exists (not Mobile)
    if self.Sidebar then
        local sidebarIconSize = Responsive:Scale(36)
        local iconContainer = Util.Create("Frame", {
            Size                   = UDim2.new(0, sidebarIconSize, 0, sidebarIconSize),
            BackgroundColor3       = Themes:Current().SurfaceAlt,
            BackgroundTransparency = 1,
            ZIndex                 = 13,
        }, self.Sidebar)
        Util.Corner(iconContainer, 10)

        local sidebarIconInner = nil
        if config.Icon then
            local ic, innerObj = Icons:Create(iconContainer, config.Icon,
                Responsive:IconSize() - 4, Themes:Current().TextMuted)
            ic.Position      = UDim2.new(0.5, -(Responsive:IconSize()-4)/2,
                                          0.5, -(Responsive:IconSize()-4)/2)
            sidebarIconInner = innerObj
        end

        tab.SidebarIconFrame = iconContainer
        tab.SidebarIconInner = sidebarIconInner
    end

    tab.ClickBtn.MouseButton1Click:Connect(function()
        self:SetActiveTab(tab)
    end)

    if #self.Tabs == 1 then
        self:SetActiveTab(tab)
    end

    if self.SearchBox then
        SearchEngine:Init(self.SearchBox, self.Tabs)
    end
    return tab
end

function Window:SetActiveTab(target)
    local theme = Themes:Current()
    for _, tab in ipairs(self.Tabs) do
        local isActive = (tab == target)
        tab:SetActive(isActive)

        -- Sidebar icon recolor
        if tab.SidebarIconFrame then
            Util.Tween(tab.SidebarIconFrame, Util.TI(0.2), {
                BackgroundColor3       = theme.SurfaceAlt,
                BackgroundTransparency = isActive and 0 or 1,
            })
        end
        if tab.SidebarIconInner then
            local targetColor = isActive and theme.Accent or theme.TextMuted
            Util.Safe(function()
                if tab.SidebarIconInner:IsA("ImageLabel") then
                    Util.Tween(tab.SidebarIconInner, Util.TI(0.2), { ImageColor3 = targetColor })
                elseif tab.SidebarIconInner:IsA("TextLabel") then
                    Util.Tween(tab.SidebarIconInner, Util.TI(0.2), { TextColor3 = targetColor })
                end
            end)
        end
    end
    self.ActiveTab = target
end

function Window:Hide()
    if not self.Visible then return end
    local sz = self.Frame.Size
    Util.Tween(self.Frame, Util.TI(0.3, Enum.EasingStyle.Quint), {
        BackgroundTransparency = 1,
        Size = UDim2.new(sz.X.Scale, sz.X.Offset * 0.90, sz.Y.Scale, sz.Y.Offset * 0.90),
    }).Completed:Connect(function()
        if self.Frame then
            self.Frame.Visible = false
            self.Frame.Size    = sz
        end
    end)
    if self.FloatBtn then self.FloatBtn.Visible = true end
    self.Visible = false
end

        local tDef   = Themes.Definitions[tName]
        local isActive = (Themes.Name == tName)
        local tBtn = Util.Create("TextButton", {
            Size             = UDim2.new(0, 1, 0, 1),  -- UIGridLayout controls actual size
            BackgroundColor3 = tDef.Surface,
            Text             = "",
            ZIndex           = 604,
            LayoutOrder      = i,
        }, themeGrid)
        Util.Corner(tBtn, 8)

        local accentBar = Util.Create("Frame", {
            Size             = UDim2.new(1, 0, 0, 4),
            Position         = UDim2.new(0, 0, 1, -4),
            BackgroundColor3 = tDef.Accent,
            ZIndex           = 605,
        }, tBtn)
        Util.Corner(accentBar, 4)

        Util.Create("TextLabel", {
            Size                   = UDim2.new(1, -8, 1, -10),
            Position               = UDim2.new(0, 6, 0, 4),
            BackgroundTransparency = 1,
            Text                   = tName,
            TextColor3             = tDef.Text,
            Font                   = isActive and Enum.Font.GothamBold or Enum.Font.Gotham,
            TextSize               = Responsive:FontSize(12),
            TextXAlignment         = Enum.TextXAlignment.Left,
            TextYAlignment         = Enum.TextYAlignment.Center,
            ZIndex                 = 605,
        }, tBtn)

        if isActive then
            Util.Stroke(tBtn, tDef.Accent, 2, 0.1)
        end

        tBtn.MouseButton1Click:Connect(function()
            Themes:Set(tName)
            SaveSystem:Set("Theme", tName)
            NotifSystem:Push({
                Title   = "Theme Changed",
                Content = tName .. " theme applied.",
                Type    = "info",
            })
            closeSettings()
        end)
    end

    -- DISPLAY section
    Util.Create("TextLabel", {
        Size                   = UDim2.new(1, 0, 0, 16),
        BackgroundTransparency = 1,
        Text                   = "DISPLAY",
        TextColor3             = theme.TextMuted,
        Font                   = Enum.Font.GothamBold,
        TextSize               = Responsive:FontSize(10),
        TextXAlignment         = Enum.TextXAlignment.Left,
        ZIndex                 = 603,
        LayoutOrder            = 2,
    }, body)

    -- Window opacity
    local opacityRow = Util.Create("Frame", {
        Size                   = UDim2.new(1, 0, 0, 0),
        AutomaticSize          = Enum.AutomaticSize.Y,
        BackgroundTransparency = 1,
        ZIndex                 = 603,
        LayoutOrder            = 3,
    }, body)
    Util.List(opacityRow, Enum.FillDirection.Vertical, 4)

    Util.Create("TextLabel", {
        Size                   = UDim2.new(1, 0, 0, 14),
        BackgroundTransparency = 1,
        Text                   = "Window Opacity",
        TextColor3             = theme.TextSub,
        Font                   = Enum.Font.Gotham,
        TextSize               = Responsive:FontSize(12),
        TextXAlignment         = Enum.TextXAlignment.Left,
        ZIndex                 = 604,
    }, opacityRow)

    -- Opacity slider (80–100%)
    local opTrackH = Responsive:Scale(8)
    local savedOpacity = SaveSystem:Get("WindowOpacity", 1.0)
    local opTrack = Util.Create("Frame", {
        Size             = UDim2.new(1, 0, 0, opTrackH),
        BackgroundColor3 = theme.Border,
        ZIndex           = 604,
    }, opacityRow)
    Util.Corner(opTrack, opTrackH/2)
    local opFill = Util.Create("Frame", {
        Size             = UDim2.new(savedOpacity, 0, 1, 0),
        BackgroundColor3 = theme.Accent,
        ZIndex           = 605,
    }, opTrack)
    Util.Corner(opFill, opTrackH/2)
    local opThumbSz = opTrackH + 6
    local opThumb = Util.Create("Frame", {
        Size             = UDim2.new(0, opThumbSz, 0, opThumbSz),
        Position         = UDim2.new(savedOpacity, -opThumbSz/2, 0.5, -opThumbSz/2),
        BackgroundColor3 = Color3.fromRGB(255,255,255),
        ZIndex           = 606,
    }, opTrack)
    Util.Corner(opThumb, opThumbSz/2)
    Util.Stroke(opThumb, theme.Accent, 1.5, 0.2)

    local opDragging = false
    local opHit = Util.Create("TextButton", {
        Size                   = UDim2.new(1, 0, 0, opTrackH + opThumbSz + 6),
        Position               = UDim2.new(0, 0, 0.5, -(opTrackH + opThumbSz + 6)/2),
        BackgroundTransparency = 1,
        Text                   = "",
        ZIndex                 = 607,
    }, opTrack)

    local function setOpacity(frac)
        frac = Util.Clamp(frac, 0, 1)
        local opacity = 0.0 + (1.0 - 0.0) * frac  -- 0% transparent to 0% transparent = fully visible always
        opFill.Size    = UDim2.new(frac, 0, 1, 0)
        opThumb.Position = UDim2.new(frac, -opThumbSz/2, 0.5, -opThumbSz/2)
        -- Map 0.5–1.0 range to 0.3–0.0 transparency
        local winTrans = Util.Clamp(1.0 - frac, 0, 0.5)
        SaveSystem:Set("WindowOpacity", frac)
        if self.Frame then
            Util.Tween(self.Frame, Util.TI(0.15), { BackgroundTransparency = winTrans })
        end
    end

    opHit.MouseButton1Down:Connect(function(x)
        opDragging = true
        local ap = opTrack.AbsolutePosition.X
        local sz = opTrack.AbsoluteSize.X
        if sz > 0 then setOpacity((x - ap) / sz) end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if not opDragging then return end
        if input.UserInputType == Enum.UserInputType.MouseMovement
        or input.UserInputType == Enum.UserInputType.Touch then
            local ap = opTrack.AbsolutePosition.X
            local sz = opTrack.AbsoluteSize.X
            if sz > 0 then setOpacity((input.Position.X - ap) / sz) end
        end
    end)
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then
            opDragging = false
        end
    end)

    -- Reset position button
    local resetBtn = Util.Create("TextButton", {
        Size             = UDim2.new(1, 0, 0, Responsive:Scale(36)),

        ZIndex           = 603,
        LayoutOrder      = 4,

    resetBtn.MouseButton1Click:Connect(function()
        local vp2   = Util.GetViewport()
        local origSz = Responsive:WindowSize()
        local cx = math.max(0, (vp2.X - origSz.X.Offset) / 2)
        local cy = math.max(0, (vp2.Y - origSz.Y.Offset) / 2)
        self.Frame.Position = UDim2.new(0, cx, 0, cy)
        SaveSystem:Set("WinPos_" .. self.Title, nil)
        NotifSystem:Push({ Title = "Position Reset", Content = "Window recentered.", Type = "info" })
    end)
    resetBtn.MouseEnter:Connect(function()
        Util.Tween(resetBtn, Util.TI(0.15), { BackgroundColor3 = theme.Surface })
    end)
    resetBtn.MouseLeave:Connect(function()
        Util.Tween(resetBtn, Util.TI(0.15), { BackgroundColor3 = theme.SurfaceAlt })
    end)

    -- Clear saves button
    local clearBtn2 = Util.Create("TextButton", {
        Size             = UDim2.new(1, 0, 0, Responsive:Scale(36)),
        BackgroundColor3 = theme.SurfaceAlt,
        Text             = "Clear All Saved Data",
        TextColor3       = theme.Danger,
        Font             = Enum.Font.GothamSemibold,
        TextSize         = Responsive:FontSize(12),
        ZIndex           = 603,
        LayoutOrder      = 5,
    }, body)
    Util.Corner(clearBtn2, 8)
    clearBtn2.MouseButton1Click:Connect(function()
        SaveSystem.Data = {}
        SaveSystem:Save()
        NotifSystem:Push({ Title = "Data Cleared", Content = "All saved data has been reset.", Type = "warning" })
        closeSettings()
    end)
    clearBtn2.MouseEnter:Connect(function()
        Util.Tween(clearBtn2, Util.TI(0.15), { BackgroundColor3 = theme.Surface })
    end)
    clearBtn2.MouseLeave:Connect(function()
        Util.Tween(clearBtn2, Util.TI(0.15), { BackgroundColor3 = theme.SurfaceAlt })
    end)

    -- Version info
    Util.Create("TextLabel", {
        Size                   = UDim2.new(1, 0, 0, 16),
        BackgroundTransparency = 1,
        Text                   = "Mixusfield v1.3.0  |  Mobile+ Ready",
        TextColor3             = theme.TextMuted,
        Font                   = Enum.Font.Gotham,
        TextSize               = Responsive:FontSize(10),
        TextXAlignment         = Enum.TextXAlignment.Center,
        ZIndex                 = 603,
        LayoutOrder            = 6,
    }, body)

    -- Animate panel in
    panel.BackgroundTransparency  = 1
    overlay.BackgroundTransparency = 1
    Util.Tween(overlay, Util.TI(0.25), { BackgroundTransparency = 0.55 })
    Util.Tween(panel, Util.TI(0.35, Enum.EasingStyle.Back), { BackgroundTransparency = 0.04 })
end

function Window:Show()
    if self.Visible then return end
    local origSz = Responsive:WindowSize()
    self.Frame.Visible              = true
    self.Frame.BackgroundTransparency = 1
    self.Frame.Size = UDim2.new(
        origSz.X.Scale, origSz.X.Offset * 0.88,
        origSz.Y.Scale, origSz.Y.Offset * 0.88
    )
    Util.Tween(self.Frame, Util.TI(0.38, Enum.EasingStyle.Back), {
        BackgroundTransparency = 0,
        Size                   = origSz,
    })
    if self.FloatBtn then self.FloatBtn.Visible = false end
    self.Visible = true
end

function Window:Minimize() self:Hide() end

function Window:Destroy()
    for _, conn in ipairs(self.Connections) do
        pcall(function() conn:Disconnect() end)
    end
    if self.Frame   then self.Frame:Destroy()   end
    if self.FloatBtn then self.FloatBtn:Destroy() end
end

local MixusfieldAPI = {}
MixusfieldAPI.__index = MixusfieldAPI

function MixusfieldAPI.Init(config)
    config = config or {}

    -- Remove existing UI
    local existing = PlayerGui:FindFirstChild("MixusfieldUI")
    if existing then existing:Destroy() end

    -- Load saves and theme
    SaveSystem:Load()
    Themes:Set(SaveSystem:Get("Theme", config.Theme or "Dark"))

    -- ScreenGui
    local screenGui = Util.Create("ScreenGui", {
        Name           = "MixusfieldUI",
        ResetOnSpawn   = false,
        ZIndexBehavior = Enum.ZIndexBehavior.Global,
        IgnoreGuiInset = true,
    })
    local ok = pcall(function() screenGui.Parent = CoreGui end)
    if not ok then screenGui.Parent = PlayerGui end

    -- Init notifications
    NotifSystem:Init(screenGui)

    -- Build window (hidden until revealed)
    local win = Window.new(screenGui, { Title = config.Title or "Mixusfield" })
    win.Frame.Visible              = false
    win.Frame.BackgroundTransparency = 1
    win.FloatBtn.Visible           = false
    win.Visible                    = false

    -- Reveal function: animate window in
    local function revealWindow()
        local origSz = Responsive:WindowSize()
        win.Frame.Visible              = true
        win.Frame.BackgroundTransparency = 1
        win.Frame.Size = UDim2.new(
            origSz.X.Scale, origSz.X.Offset * 0.88,
            origSz.Y.Scale, origSz.Y.Offset * 0.88
        )
        Util.Tween(win.Frame, Util.TI(0.42, Enum.EasingStyle.Back), {
            BackgroundTransparency = 0,
            Size                   = origSz,
        })
        win.Visible = true
    end

    -- FIX: Correct sequencing — popup first, then key system, then window
    local popupDuration = 0
    if config.StartupPopup and config.StartupPopup.Enabled then
        popupDuration = (config.StartupPopup.Duration or 4) + 0.6
        task.delay(0.3, function()
            StartupPopup:Show(screenGui, config.StartupPopup)
        end)
    end

    task.delay(popupDuration + 0.2, function()
        if config.KeySystem then
            -- Window only appears via onSuccess (valid key submitted)
            KeySystem:Create(
                screenGui,
                config.KeySystem,
                config.KeySystem.Callback or function() return true end,
                revealWindow
            )
        else
            revealWindow()
        end
    end)

    -- Theme change listener
    Themes:OnChange(function(t)
        Util.Safe(function()
            win.Frame.BackgroundColor3      = t.Background
            win.Topbar.BackgroundColor3     = t.TopbarBg
            win.Sidebar.BackgroundColor3    = t.SidebarBg
            win.TabBar.BackgroundColor3     = t.Surface
            win.ContentArea.BackgroundColor3 = t.Background
            win.TitleLbl.TextColor3         = t.Text
        end)
    end)

    local api = setmetatable({}, MixusfieldAPI)
    api._screenGui = screenGui
    api._window    = win
    return api
end

function MixusfieldAPI:AddTab(config)    return self._window:AddTab(config) end
function MixusfieldAPI:Notify(config)   NotifSystem:Push(config) end
function MixusfieldAPI:SetTheme(name)   Themes:Set(name); SaveSystem:Set("Theme", name) end
function MixusfieldAPI:GetTheme()       return Themes.Name end
function MixusfieldAPI:Destroy()
    self._window:Destroy()
    if self._screenGui then self._screenGui:Destroy() end
end
